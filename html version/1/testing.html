Bubble Sort (Very Simple)

• Take two numbers
• Compare them
• If the left number is bigger, swap them
• Bigger values slowly move right
• When no swaps happen → sorted ✅
Click "Next Step" to start
#include <iostream>
using namespace std;
int main() {
int arr[] = {5, 3, 8, 4, 2};
int n = 5;
for (int i = 0; i < n - 1; i++) {
for (int j = 0; j < n - i - 1; j++) {
if (arr[j] > arr[j + 1]) {
swap(arr[j], arr[j + 1]);
}
}
}
return 0;
}
